province_event = { #Paper for Savoy Area
	id = flaprices.4
	title = flaprices.4.t
	desc = flaprices.4.d
	picture = TRADEGOODS_eventPicture
	fire_only_once = yes
	major = yes
	
	trigger = {
		area = savoy_dauphine_area
		#savoy_dauphine_area = { #Does this work?
		#	NOT = { trade_goods = paper }
		#}
		NOT = { is_year = 1550 }
		is_state = yes
		development = 15
		mercantilism_institution = 100
		owner = { has_institution = mercantilism_institution } #Keep this?
		OR = { #Ruler requirement
			owner = { adm = 5 }
			owner = { dip = 5 }
		}
		OR = { #Trade Goods required
			trade_goods = fish
			trade_goods = livestock
			trade_goods = grain
			trade_goods = wool
			AND = { #cloth
				trade_goods = cloth
				development = 20
			}
			AND = { #wine
				trade_goods = wine
				development = 20
			}
		}
	}
	
	mean_time_to_happen = {
		months = 420 #35 years
	}
	
	option = {
		name = flaprices.4.a
		# savoy_dauphine_area = {
			# random_owned_province = {
				# limit = {
					# NOT = { trade_goods = paper }
					# is_state = yes
					# development = 10
				# }
				# change_trade_goods = paper
				# add_base_production = 2
			# }
		# }
		change_trade_goods = paper
		add_base_production = 2
	}
}
