	title = "flagreatproject.2.t"
	desc = "flagreatproject.2.d"
	picture = HARBOUR_eventPicture
	
	fire_only_once = yes

	trigger = {
		owns = 358
		NOT = { has_country_modifier = alx_great_library_perma_modifier }
		358 = {
			has_great_project = great_library
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}

	option = {

		owner = { add_prestige = 30 }

			duration = -1 
		}
		
	}
}
