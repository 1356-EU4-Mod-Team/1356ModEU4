#Event that gives the nations the option of joining the HRE or getting a claim on all of France
country_event = {
	id = fre_disaster_civil_war.3
	title = fre_disaster_civil_war.3.t
	desc = fre_disaster_civil_war.3.desc
	picture = HRE_eventPicture
	
	major = yes
	
	trigger = {
		has_global_flag = 1356_french_civil_war
		technology_group = western
		culture_group = french
		OR = {
			current_age = age_of_feudalism
			NOT = { is_year = 1480 }
		}
		is_part_of_hre = no
		NOT = { has_global_flag = emperor_declined_offer_to_french_dukes }
		NOT = { has_country_flag = fre_minor_declined_hre_offer_flag }
		NOT = { tag = FRA }
		NOT = { tag = ENG }
		NOT = { tag = SPA }
		NOT = { tag = ARA }
		NOT = { tag = NAV }
		NOT = { tag = DAN }
		NOT = { tag = KNI }
		NOT = { tag = CYP }
		NOT = { tag = CIC }
		NOT = { tag = KOJ }
		OR = {
			AND = {
				tag = BUR
				NOT = { num_of_subjects = 2 }
				NOT = { num_of_states = 4 }
			}
			NOT = { tag = BUR }
		}
		is_subject = no
		NOT = { num_of_states = 6 }
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = { #We stay independent!
		name = fre_disaster_civil_war.3.a
		ai_chance = { factor = 25 }
		france_region = {
			add_permanent_claim = ROOT
		}
		add_prestige = 5
		add_legitimacy = 5
		ROOT = { 
			set_country_flag = fre_minor_declined_hre_offer_flag
		}
		emperor = {
			country_event = { 
			id = fre_disaster_civil_war.6
			days = 10
			}
		}
	}
	
	option = {
		name = fre_disaster_civil_war.3.b
		ai_chance = { factor = 75 }
		ROOT = {
			set_in_empire = yes
			capital_scope = { 
				set_in_empire = yes
			}
		}
		emperor = {		
			country_event = { 
				id = fre_disaster_civil_war.5
				days = 10
			}
		}
	}
	
}
