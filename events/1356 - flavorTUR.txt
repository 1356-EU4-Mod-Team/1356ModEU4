	#major = yes

			set_global_flag = allow_splendor_for_ai

		hidden_effect = {
			HUN = {
				add_opinion = {
					who = SER
					modifier = 1356_past_transgressions
				}
				reverse_add_opinion = {
					who = SER
					modifier = 1356_past_transgressions
				}	
				add_opinion = {
					who = BUL
					modifier = 1356_past_transgressions
				}
				reverse_add_opinion = {
					who = BUL
					modifier = 1356_past_transgressions
				}					
			}
			SER = {
				add_opinion = {
					who = BUL
					modifier = 1356_past_cordial_relations
				}
				reverse_add_opinion = {
					who = BUL
					modifier = 1356_past_cordial_relations
				}
				add_trust = {
					who = BUL	
					value = 30
					mutual = yes
				}	
				remove_historical_friend = BUL
				BUL = {
					remove_historical_friend = SER
				}	
			}	
			BUL = {
				add_opinion = {
					who = BYZ
					modifier = 1356_past_cordial_relations
				}
				reverse_add_opinion = {
					who = BYZ
					modifier = 1356_past_cordial_relations
				}
				add_trust = {
					who = BYZ	
					value = 15
					mutual = yes
				}	
				remove_historical_friend = BYZ
				BYZ = {
					remove_historical_friend = BUL

			}				
		}	
	}
