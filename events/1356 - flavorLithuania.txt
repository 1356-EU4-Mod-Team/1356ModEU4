#Union of Krewo (Poland) (Poland is a player version & Lith is an AI)
country_event = {
	id = flaflavorlith.6
	title = flaflavorlith.6.t
	desc = flaflavorlith.6.d

	picture = GOOD_WITH_MONARCH_eventPicture
	
	fire_only_once = yes
	major = yes
	
	trigger = {
		tag = POL
		is_subject = no
		ai = no
		has_regency = no
		is_female = yes
		LIT  = {
			ai = yes
			exists = yes
			is_subject = no
			was_player = no
			NOT = { junior_union_with = POL }
			is_female = no
		}
		NOT = { is_year = 1450 }
	}
	
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.1
			is_year = 1370
		}
		modifier = {
			factor = 0.01
			is_year = 1375
		}
		modifier = {
			factor = 0.001
			is_year = 1385
		}
	}
	
	option = {
		name = flaflavorlith.6.a #We must retain the Jagiellon Dynasty
		ai_chance = { 
			factor = 80 
		}
		create_union = LIT
		#Audax Validator "." Ignore_NEXT
		set_dynasty = "Jagiellon"
		define_general = {
			name = "Jogaila Jagiellon"
			fire = 3
			shock = 3	
			manuever = 3
			siege = 1
			female = no
		}		
		define_consort = {
			name = "Jogaila"
			dynasty = "Jagiellon"
			culture = lithuanian
			adm = 5
			dip = 3
			mil = 5
			female = no
		}
		add_historical_friend = LIT
		LIT = {
			add_historical_friend = POL
		}
		LIT = {
			change_religion = catholic
			every_owned_province = {
				limit = {
					religion = baltic_pagan_reformed
				}		
				change_religion = catholic
				add_province_modifier = {
					name = forced_conversion
					duration = 4800
				}
			}
		}	
		remove_heir = yes
	}
	
	option = {
		name = flaflavorlith.6.b #Stay Romuva.
		ai_chance = { 
			factor = 20
		}
		LIT = {
			add_opinion = {
				who = ROOT
				modifier = lit_denied_birthright
			}
		}
	}
}
