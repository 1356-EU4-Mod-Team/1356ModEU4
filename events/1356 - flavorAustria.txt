country_event = { #The Purchase of Breisgau
	id = flavor_1356_hab.5
	title = flavor_1356_hab.5.t 
	desc = flavor_1356_hab.5.d 
	picture = DIPLOMACY_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = HAB
		STY = {
			is_subject_of = HAB
		}
		72 = {
			NOT = {
				owned_by = HAB
				owned_by = STY 
			}
			NOT = {
				is_capital = yes
			}
		}
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	option = { #buy it
		name = flavor_1356_hab.5.a 
		72 = {
			owner = {
				country_event = {
					id = flavor_1356_hab.6
				}
			}
		}
		add_treasury = -125
	}
	
	option = { #don't buy it
		name = flavor_1356_hab.5.b 
		add_prestige = 5
	}
}
