				}

			}
		}

			limit = {
				CXI = {

				}
			}

			}

					was_player = no
				}

				inherit = SHU
			}

		if = {
			limit = {
				QIC = {
					was_player = no
				}
			}
			MNG = {
				inherit = QIC
			}
		}

			#		was_player = no
			#	}

					subject = MIN

			xinan_region = {
				add_core = MNG
