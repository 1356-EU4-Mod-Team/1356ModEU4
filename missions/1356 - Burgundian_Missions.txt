	slot = 1

	potential = { 
		OR = {

	1356_BUR_FLA_union = { #We were promised a marriage with the Duchess of Flanders! We should rule it in union, together. As husband and Wife!
		icon = mission_1356_flanders
		required_missions = { }
		completed_by = 1384.1.30
		provinces_to_highlight = {
			owner = {
				OR = {
					tag = FLA
					junior_union_with = FLA
				}
			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}	
		trigger = {
			OR = {
				#ROOT must either have 200 opinion and royal marriage with FLA
				AND = {
					FLA = {
						has_opinion = {
							who = ROOT
							value = 200
						}
					}
					marriage_with = FLA
				}
				#Or ROOT must have alliance with FLA and 70 trust
				AND = {
					alliance_with = FLA
					trust = { who = FLA value = 70 }
				}
				#Bypass if owning some of Flanders
				num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
					value = 2
					area = flanders_area
				}
				#Bypass if Flanders does not exist, or either nation is invalid for unions
				NOT = { exists = FLA }
				NOT = { valid_for_personal_unions_trigger = yes }
				FLA = { NOT = { valid_for_personal_unions_trigger = yes } }
			}
		}
		effect = {
			#If FLA exists and is valid for personal unions it gets union proposal event
			if = {
				limit = {
					valid_for_personal_unions_trigger = yes
					FLA = {
						exists = yes
						is_subject = no
						valid_for_personal_unions_trigger = yes
					}
				}
				FLA = {
					country_event = {
						id = flavor_fla_bur.5
						days = 25
						tooltip = 1356_BUR_FLA_union_tooltip.1
					}
				}
			}
			#Else if already owning Flanders, ROOT gets some prestige instead
			else_if = {
				limit = {
					num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
						value = 3
						area = flanders_area
					}
				}
				add_prestige = 20
			}
			#Else ROOT gets claim on Flanders
			else = {
				flanders_area = {
					limit = {
						NOT = { is_permanent_claim = ROOT }
						NOT = { is_core = ROOT	}
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}

						NOT = { valid_for_personal_unions_trigger = yes }

		}
		
		effect = {
			#If HAI exists and is valid for union, HAI gets an event demanding submission

				limit = {
					valid_for_personal_unions_trigger = yes

						exists = yes
						valid_for_personal_unions_trigger = yes
						NOT = { is_subject_of = ROOT }
						OR = {
							owns_or_non_sovereign_subject_of = 1743	#Cambrésis
							owns_or_non_sovereign_subject_of = 91	#Hainaut
						}
					}
				}
				HAI = {
					country_event = {
						id = flavor_fla_bur.40
						days = 25
						tooltip = 1356_BUR_HAI_union_tooltip.1
					}
				}
			}
			#If HAI is a subject of ROOT and only controls original HAI cores then integrate HAI into ROOT
			else_if = {
				limit = {
					HAI = { is_subject_of = ROOT }
					NOT = {
						HAI = {
							any_owned_province = {
								NOT = {
									province_id = 91
									province_id = 1743
								}
							}
						}
					}
				}
				inherit = HAI
			}
			#If already owning original HAI cores then get some prestige
			else_if = {
				limit = {
					AND = {
						owns_or_non_sovereign_subject_of = 1743	#Cambrésis
						owns_or_non_sovereign_subject_of = 91	#Hainaut
					}					
				}
				add_prestige = 5
			}
			#Else get permanent claims on original HAI cores
			else = {
				1743 = {
					limit = {
						NOT = { is_core = root }
						NOT = { is_permanent_claim = root }
						NOT = { owns_or_non_sovereign_subject_of = 1743 }
					}
					add_permanent_claim = root
				}
				91 = {
					limit = {
						NOT = { is_core = root }
						NOT = { is_permanent_claim = root }
						NOT = { owns_or_non_sovereign_subject_of = 91 }
					}
					add_permanent_claim = root
				}
			}
		}
